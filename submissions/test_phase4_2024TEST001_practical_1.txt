# Temperature Monitoring System - Practical 1

## Overview
This Python program monitors temperature data and provides analysis functionality.

## Implementation

```python
import datetime
import statistics

class TemperatureMonitor:
    """A class to monitor and analyze temperature readings."""
    
    def __init__(self):
        self.readings = []
        self.timestamps = []
    
    def add_reading(self, temp, timestamp=None):
        """Add a temperature reading with timestamp."""
        if timestamp is None:
            timestamp = datetime.datetime.now()
        self.readings.append(temp)
        self.timestamps.append(timestamp)
    
    def get_average(self):
        """Calculate average temperature."""
        if not self.readings:
            return 0
        return statistics.mean(self.readings)
    
    def get_max(self):
        """Get maximum temperature."""
        return max(self.readings) if self.readings else 0
    
    def get_min(self):
        """Get minimum temperature."""
        return min(self.readings) if self.readings else 0
    
    def get_report(self):
        """Generate a temperature report."""
        report = {
            'average': self.get_average(),
            'max': self.get_max(),
            'min': self.get_min(),
            'total_readings': len(self.readings)
        }
        return report

# Main program
if __name__ == "__main__":
    monitor = TemperatureMonitor()
    
    # Add some test readings
    monitor.add_reading(22.5)
    monitor.add_reading(23.1)
    monitor.add_reading(21.8)
    monitor.add_reading(24.2)
    monitor.add_reading(22.9)
    
    # Generate and print report
    report = monitor.get_report()
    print("Temperature Monitoring Report")
    print("-" * 30)
    print(f"Average: {report['average']:.2f}°C")
    print(f"Maximum: {report['max']:.2f}°C")
    print(f"Minimum: {report['min']:.2f}°C")
    print(f"Total Readings: {report['total_readings']}")
```
