```python
# Temperature Monitoring System
# Student: Rahul Kumar (2024CS050)

def collect_temperature_data(num_readings=10):
    """
    Collects temperature readings from sensor
    Returns list of temperature values
    """
    import random
    temperatures = []
    
    print(f"Collecting {num_readings} temperature readings...")
    print("-" * 50)
    
    for i in range(1, num_readings + 1):
        # Simulate sensor reading
        temp = round(random.uniform(18.0, 32.0), 1)
        temperatures.append(temp)
        print(f"Reading {i:2d}: {temp}°C")
    
    return temperatures

def analyze_temperature(temps):
    """
    Analyzes temperature data and returns statistics
    """
    if not temps:
        return None
    
    stats = {
        'count': len(temps),
        'average': sum(temps) / len(temps),
        'max': max(temps),
        'min': min(temps),
        'range': max(temps) - min(temps)
    }
    
    return stats

def display_results(stats):
    """
    Displays analysis results
    """
    print("\n" + "=" * 50)
    print("TEMPERATURE ANALYSIS RESULTS")
    print("=" * 50)
    print(f"Total Readings: {stats['count']}")
    print(f"Average: {stats['average']:.2f}°C")
    print(f"Maximum: {stats['max']:.1f}°C")
    print(f"Minimum: {stats['min']:.1f}°C")
    print(f"Range: {stats['range']:.1f}°C")
    
    # Status check
    avg = stats['average']
    if avg < 20:
        status = "⚠️ BELOW NORMAL"
    elif avg > 30:
        status = "⚠️ ABOVE NORMAL"
    else:
        status = "✅ NORMAL"
    
    print(f"\nStatus: {status}")
    print("=" * 50)

def main():
    """Main program"""
    print("Temperature Sensor Testing Program")
    print("=" * 50)
    
    # Collect data
    readings = collect_temperature_data(10)
    
    # Analyze data
    results = analyze_temperature(readings)
    
    # Display results
    if results:
        display_results(results)
    else:
        print("Error: No data to analyze")

if __name__ == "__main__":
    main()
```
