class SimpleCalculator {
    constructor() {
        this.memory = 0;
        this.history = [];
    }
    
    // Basic arithmetic operations
    add(a, b) {
        const result = parseFloat(a) + parseFloat(b);
        this.history.push(`${a} + ${b} = ${result}`);
        return result;
    }
    
    subtract(a, b) {
        const result = parseFloat(a) - parseFloat(b);
        this.history.push(`${a} - ${b} = ${result}`);
        return result;
    }
    
    multiply(a, b) {
        const result = parseFloat(a) * parseFloat(b);
        this.history.push(`${a} × ${b} = ${result}`);
        return result;
    }
    
    divide(a, b) {
        if (parseFloat(b) === 0) {
            throw new Error("Cannot divide by zero");
        }
        const result = parseFloat(a) / parseFloat(b);
        this.history.push(`${a} ÷ ${b} = ${result}`);
        return result;
    }
    
    // Advanced operations
    power(base, exponent) {
        const result = Math.pow(parseFloat(base), parseFloat(exponent));
        this.history.push(`${base}^${exponent} = ${result}`);
        return result;
    }
    
    squareRoot(number) {
        if (parseFloat(number) < 0) {
            throw new Error("Cannot calculate square root of negative number");
        }
        const result = Math.sqrt(parseFloat(number));
        this.history.push(`√${number} = ${result}`);
        return result;
    }
    
    // Memory operations
    memoryStore(value) {
        this.memory = parseFloat(value);
        return `Stored ${value} in memory`;
    }
    
    memoryRecall() {
        return this.memory;
    }
    
    memoryClear() {
        this.memory = 0;
        return "Memory cleared";
    }
    
    // History operations
    getHistory() {
        return this.history.join('\n');
    }
    
    clearHistory() {
        this.history = [];
        return "History cleared";
    }
}

// Test cases
const calc = new SimpleCalculator();

console.log("Basic Operations:");
console.log(calc.add(10, 5));        // 15
console.log(calc.subtract(10, 3));   // 7
console.log(calc.multiply(4, 5));    // 20
console.log(calc.divide(20, 4));     // 5

console.log("\nAdvanced Operations:");
console.log(calc.power(2, 3));       // 8
console.log(calc.squareRoot(16));    // 4

console.log("\nMemory Operations:");
console.log(calc.memoryStore(42));
console.log(calc.memoryRecall());
console.log(calc.memoryClear());

console.log("\nHistory:");
console.log(calc.getHistory());

module.exports = SimpleCalculator;
