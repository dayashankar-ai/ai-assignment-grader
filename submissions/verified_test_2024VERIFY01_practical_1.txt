/**
 * Simple Calculator Implementation - Practical 1
 * This is a comprehensive calculator with memory and history features
 */

class Calculator {
    constructor() {
        this.memory = 0;
        this.history = [];
    }

    add(a, b) {
        const result = a + b;
        this.history.push(`${a} + ${b} = ${result}`);
        return result;
    }

    subtract(a, b) {
        const result = a - b;
        this.history.push(`${a} - ${b} = ${result}`);
        return result;
    }

    multiply(a, b) {
        const result = a * b;
        this.history.push(`${a} ร ${b} = ${result}`);
        return result;
    }

    divide(a, b) {
        if (b === 0) throw new Error("Division by zero");
        const result = a / b;
        this.history.push(`${a} รท ${b} = ${result}`);
        return result;
    }

    power(base, exp) {
        const result = Math.pow(base, exp);
        this.history.push(`${base}^${exp} = ${result}`);
        return result;
    }

    sqrt(n) {
        if (n < 0) throw new Error("Cannot sqrt negative number");
        return Math.sqrt(n);
    }

    memoryStore(val) { this.memory = val; }
    memoryRecall() { return this.memory; }
    memoryClear() { this.memory = 0; }

    getHistory() {
        return this.history.join('\n');
    }
}

// Test the calculator
const calc = new Calculator();
console.log("5 + 3 =", calc.add(5, 3));
console.log("10 - 4 =", calc.subtract(10, 4));
console.log("6 * 7 =", calc.multiply(6, 7));
console.log("20 / 4 =", calc.divide(20, 4));
console.log("2^8 =", calc.power(2, 8));
console.log("History:", calc.getHistory());
