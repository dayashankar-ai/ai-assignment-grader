# Practical 1: Prompt Engineering - Temperature Testing
# By: Emily Davis (2024CS020)

from anthropic import Anthropic

# Initialize API client
api_key = "test-key"
client = Anthropic(api_key=api_key)

print("=== Temperature Experiment ===\n")

# Test prompt
test_prompt = "Explain recursion in programming"

# Test three temperature values
temperatures = [0.0, 0.5, 1.0]

for temp in temperatures:
    print(f"\nTemperature: {temp}")
    print("-" * 40)
    
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=100,
        temperature=temp,
        messages=[{"role": "user", "content": test_prompt}]
    )
    
    print(f"Output: {response.content[0].text}\n")

# Learning Summary:
print("\n=== What I Learned ===")
print("1. Temperature 0.0 = Deterministic, same output every time")
print("2. Temperature 0.5 = Balanced creativity and consistency")  
print("3. Temperature 1.0 = Maximum creativity, varied outputs")
print("\nBest practice: Use low temp for facts, high temp for creativity")
