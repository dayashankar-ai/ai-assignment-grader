import datetime
import statistics

class TemperatureMonitor:
    """
    A class to monitor temperature readings over time.
    Stores readings with timestamps and provides statistical analysis.
    """
    
    def __init__(self):
        # Lists to store our data
        self.readings = []
        self.timestamps = []
        print("Temperature monitor initialized!")
    
    def add_reading(self, temperature, timestamp=None):
        """
        Add a new temperature reading.
        
        Args:
            temperature (float): The temperature value in Celsius
            timestamp (datetime): When the reading was taken (defaults to now)
        """
        # If no timestamp provided, use current time
        if timestamp is None:
            timestamp = datetime.datetime.now()
        
        # Store the reading
        self.readings.append(temperature)
        self.timestamps.append(timestamp)
        
        print(f"Added reading: {temperature}°C at {timestamp}")
    
    def get_average(self):
        """Calculate and return the average temperature."""
        if not self.readings:
            return 0.0
        
        # I learned about the statistics module for this!
        return statistics.mean(self.readings)
    
    def get_max(self):
        """Return the highest temperature recorded."""
        if not self.readings:
            return 0.0
        return max(self.readings)
    
    def get_min(self):
        """Return the lowest temperature recorded."""
        if not self.readings:
            return 0.0
        return min(self.readings)
    
    def get_standard_deviation(self):
        """Calculate standard deviation to see temperature variance."""
        if len(self.readings) < 2:
            return 0.0
        return statistics.stdev(self.readings)
    
    def get_report(self):
        """
        Generate a comprehensive temperature report.
        
        Returns:
            dict: Dictionary containing all statistics
        """
        if not self.readings:
            return {
                'message': 'No readings available',
                'count': 0
            }
        
        report = {
            'average': round(self.get_average(), 2),
            'maximum': round(self.get_max(), 2),
            'minimum': round(self.get_min(), 2),
            'std_deviation': round(self.get_standard_deviation(), 2),
            'total_readings': len(self.readings),
            'first_reading': str(self.timestamps[0]),
            'last_reading': str(self.timestamps[-1])
        }
        
        return report

# Main program execution
if __name__ == "__main__":
    # Create a new monitor instance
    monitor = TemperatureMonitor()
    
    # Add some test readings
    # I used realistic indoor temperature values
    print("\n--- Adding Temperature Readings ---")
    monitor.add_reading(22.5)
    monitor.add_reading(23.1)
    monitor.add_reading(21.8)
    monitor.add_reading(24.2)
    monitor.add_reading(22.9)
    monitor.add_reading(23.5)
    monitor.add_reading(22.0)
    
    # Generate and print the report
    print("\n--- Temperature Report ---")
    report = monitor.get_report()
    
    print("=" * 40)
    print("TEMPERATURE MONITORING SYSTEM REPORT")
    print("=" * 40)
    print(f"Total Readings: {report['total_readings']}")
    print(f"Average Temperature: {report['average']}°C")
    print(f"Maximum Temperature: {report['maximum']}°C")
    print(f"Minimum Temperature: {report['minimum']}°C")
    print(f"Standard Deviation: {report['std_deviation']}°C")
    print(f"First Reading: {report['first_reading']}")
    print(f"Last Reading: {report['last_reading']}")
    print("=" * 40)
    
    # Additional analysis
    temp_range = report['maximum'] - report['minimum']
    print(f"\nTemperature Range: {temp_range}°C")
    
    if report['std_deviation'] < 1.0:
        print("Temperature stability: GOOD (low variation)")
    else:
        print("Temperature stability: MODERATE (some variation)")
