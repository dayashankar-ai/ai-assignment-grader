=== TEMPERATURE TESTING PROGRAM ===

```python
import random
import statistics

def read_temperature():
    """Simulate reading from temperature sensor"""
    return round(random.uniform(22.0, 28.0), 2)

def main():
    print("Temperature Monitoring System")
    print("=" * 40)
    
    temperatures = []
    
    # Collect 10 readings
    print("\nCollecting readings...")
    for i in range(1, 11):
        temp = read_temperature()
        temperatures.append(temp)
        status = "NORMAL" if 23 <= temp <= 27 else "CHECK"
        print(f"Reading {i}: {temp}°C [{status}]")
    
    # Calculate statistics
    avg_temp = statistics.mean(temperatures)
    max_temp = max(temperatures)
    min_temp = min(temperatures)
    
    print("\n" + "=" * 40)
    print("STATISTICS")
    print("=" * 40)
    print(f"Average: {avg_temp:.2f}°C")
    print(f"Maximum: {max_temp:.2f}°C")
    print(f"Minimum: {min_temp:.2f}°C")
    print(f"Range: {max_temp - min_temp:.2f}°C")
    
    # Overall assessment
    if avg_temp < 23:
        print("\nStatus: BELOW OPTIMAL - Check heating")
    elif avg_temp > 27:
        print("\nStatus: ABOVE OPTIMAL - Check cooling")
    else:
        print("\nStatus: OPTIMAL RANGE")

if __name__ == "__main__":
    main()
```
