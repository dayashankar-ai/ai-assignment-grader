from anthropic import Anthropic

client = Anthropic(api_key="my-api-key")

# Testing different temperature values
def test_temperatures():
    """Experiment with temperature parameter"""
    
    temperatures = [0.0, 0.5, 1.0]
    prompt = "Explain machine learning in simple terms"
    
    print("Temperature Experiment Results:\n")
    
    for temp in temperatures:
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=150,
            temperature=temp,
            messages=[{
                "role": "user",
                "content": prompt
            }]
        )
        
        print(f"Temperature: {temp}")
        print(f"Response: {response.content[0].text}")
        print("-" * 50)

# Run the experiment
test_temperatures()

# Learning Report:

# What I Learned:
# - Temperature=0.0 gives consistent, deterministic outputs
# - Temperature=0.5 provides a good balance between consistency and creativity
# - Temperature=1.0 produces more varied and creative responses

# - Lower temperature is better for factual tasks
