## Python Code

```python
import datetime
import statistics

class TemperatureMonitor:
    """Monitor and analyze temperature readings."""
    
    def __init__(self):
        self.readings = []
        self.timestamps = []
        print("Temperature monitor initialized!")
    
    def add_reading(self, temp, timestamp=None):
        """Add a temperature reading with timestamp."""
        if timestamp is None:
            timestamp = datetime.datetime.now()
        
        self.readings.append(temp)
        self.timestamps.append(timestamp)
        print(f"Added: {temp}째C at {timestamp}")
    
    def get_average(self):
        """Calculate average temperature."""
        if not self.readings:
            return 0.0
        return statistics.mean(self.readings)
    
    def get_max(self):
        """Get maximum temperature."""
        return max(self.readings) if self.readings else 0.0
    
    def get_min(self):
        """Get minimum temperature."""
        return min(self.readings) if self.readings else 0.0
    
    def get_report(self):
        """Generate temperature report."""
        return {
            'average': round(self.get_average(), 2),
            'max': round(self.get_max(), 2),
            'min': round(self.get_min(), 2),
            'total': len(self.readings)
        }

# Main execution
if __name__ == "__main__":
    monitor = TemperatureMonitor()
    
    # Test with various temperatures
    monitor.add_reading(22.5)
    monitor.add_reading(23.1)
    monitor.add_reading(21.8)
    monitor.add_reading(24.2)
    monitor.add_reading(22.9)
    
    # Print report
    report = monitor.get_report()
    print("\n=== Temperature Report ===")
    print(f"Average: {report['average']}째C")
    print(f"Max: {report['max']}째C")
    print(f"Min: {report['min']}째C")
    print(f"Total: {report['total']}")
```
