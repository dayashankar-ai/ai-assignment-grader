```python
# Temperature Sensor Testing Program
# Author: Neha Patel (2024CS060)

import random

def get_sensor_reading():
    """Simulate temperature sensor reading"""
    return round(random.uniform(22.0, 28.0), 2)

def main():
    print("=" * 60)
    print("TEMPERATURE SENSOR TESTING PROGRAM")
    print("=" * 60)
    print()
    
    # Collect 10 readings
    readings = []
    print("Collecting 10 temperature readings:\n")
    
    for i in range(1, 11):
        temp = get_sensor_reading()
        readings.append(temp)
        
        # Display reading with status
        if temp < 23:
            status = "LOW"
        elif temp > 27:
            status = "HIGH"
        else:
            status = "NORMAL"
        
        print(f"Reading {i:2d}: {temp:5.2f}°C - Status: {status}")
    
    # Calculate statistics
    print("\n" + "-" * 60)
    print("STATISTICAL ANALYSIS")
    print("-" * 60)
    
    average = sum(readings) / len(readings)
    maximum = max(readings)
    minimum = min(readings)
    temp_range = maximum - minimum
    
    print(f"Total Readings: {len(readings)}")
    print(f"Average Temperature: {average:.2f}°C")
    print(f"Maximum Temperature: {maximum:.2f}°C")
    print(f"Minimum Temperature: {minimum:.2f}°C")
    print(f"Temperature Range: {temp_range:.2f}°C")
    
    # Overall status
    print("\n" + "-" * 60)
    if average < 23:
        print("Overall Status: BELOW OPTIMAL RANGE")
        print("Recommendation: Check heating system")
    elif average > 27:
        print("Overall Status: ABOVE OPTIMAL RANGE")
        print("Recommendation: Check cooling system")
    else:
        print("Overall Status: WITHIN OPTIMAL RANGE")
        print("Recommendation: System operating normally")
    
    print("-" * 60)

if __name__ == "__main__":
    main()
```
