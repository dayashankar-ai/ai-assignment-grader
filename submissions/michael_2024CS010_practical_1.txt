from anthropic import Anthropic
import os

# Initialize client
client = Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))

def experiment_with_temperature():
    """
    Testing how temperature affects AI responses
    """
    
    prompt = "Write a short story about a robot learning to paint"
    temperatures = [0.0, 0.5, 1.0]
    
    results = {}
    
    for temp in temperatures:
        print(f"\n{'='*60}")
        print(f"Testing Temperature: {temp}")
        print('='*60)
        
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=200,
            temperature=temp,
            messages=[{
                "role": "user",
                "content": prompt
            }]
        )
        
        results[temp] = response.content[0].text
        print(f"Response:\n{results[temp]}")
    
    return results

# Run experiments
print("Starting Temperature Experiments...")
experiment_results = experiment_with_temperature()

# Analysis
print("\n\n" + "="*60)
print("ANALYSIS AND LEARNING REPORT")
print("="*60)

print("""
Key Findings:

1. Temperature = 0.0 (Deterministic)
   - Very consistent responses
   - Same prompt gives nearly identical output
   - Best for factual, consistent information
   - Less creative but more predictable

2. Temperature = 0.5 (Balanced)
   - Good balance between creativity and consistency
   - Responses vary but stay relevant
   - Suitable for most general tasks
   - Recommended default for mixed purposes

3. Temperature = 1.0 (Creative)
   - Highly varied and creative responses
   - Can be unpredictable
   - Great for creative writing and brainstorming
   - May lose focus on specific requirements

Practical Applications:
- Use temp=0.0 for: Code generation, factual Q&A, documentation
- Use temp=0.5 for: General chatbots, balanced content
- Use temp=1.0 for: Creative writing, idea generation, artistic content

Challenges Faced:
- Understanding when to use each temperature setting
- Balancing creativity vs accuracy needs
- Testing multiple times to see consistent patterns

What I Learned:

- There's no "best" temperature - it depends on the use case
- Experimentation is key to understanding the effects
- Small changes in temperature can have big impacts on output

Future Improvements:
- Try intermediate temperatures (0.3, 0.7, etc.)
- Test with different types of prompts

""")
