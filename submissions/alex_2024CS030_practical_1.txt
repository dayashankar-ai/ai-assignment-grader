=== TEMPERATURE TESTING PROGRAM ===

```python
# Temperature Testing Program
# This program tests temperature sensor by taking 10 readings

def read_temperature():
    """Simulates reading from temperature sensor"""
    import random
    return round(random.uniform(20.0, 30.0), 2)

def analyze_readings(readings):
    """Analyzes temperature readings"""
    avg = sum(readings) / len(readings)
    max_temp = max(readings)
    min_temp = min(readings)
    
    print(f"\n=== Temperature Analysis ===")
    print(f"Average Temperature: {avg:.2f}째C")
    print(f"Maximum Temperature: {max_temp:.2f}째C")
    print(f"Minimum Temperature: {min_temp:.2f}째C")
    
    # Check if within normal range
    if 18 <= avg <= 32:
        print("Status: NORMAL")
    else:
        print("Status: WARNING - Temperature out of range")
    
    return avg, max_temp, min_temp

def main():
    print("Starting Temperature Sensor Test...")
    print("Taking 10 readings...\n")
    
    readings = []
    for i in range(1, 11):
        temp = read_temperature()
        readings.append(temp)
        print(f"Reading {i}: {temp}째C")
    
    analyze_readings(readings)

if __name__ == "__main__":
    main()
```
