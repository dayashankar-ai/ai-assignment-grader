=== TEMPERATURE TESTING PROGRAM ===

```python
import random

def main():
    print("Temperature Sensor Test")
    
    readings = []
    for i in range(10):
        temp = round(random.uniform(20, 30), 1)
        readings.append(temp)
        print(f"Reading {i+1}: {temp}째C")
    
    avg = sum(readings) / len(readings)
    print(f"\nAverage: {avg:.1f}째C")
    print(f"Max: {max(readings):.1f}째C")
    print(f"Min: {min(readings):.1f}째C")

if __name__ == "__main__":
    main()
```
